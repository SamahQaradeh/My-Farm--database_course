create table animal(
ANIMAL_ID NUMBER(20) NOT NULL,
ANIMAL_NAME VARCHAR2(30),
QUANTITY NUMBER(20),
SALARY NUMBER(20),
SLEEPHOUR NUMBER(20),
COLOR VARCHAR2(30),
E_ID  NUMBER(20),
s_ID  NUMBER(20),
PRIMARY KEY(ANIMAL_ID),
FOREIGN KEY(E_ID) REFERENCES EMPLOYEE(EMP_ID)
ON DELETE CASCADE,
FOREIGN KEY(s_ID) REFERENCES storehouse(st_ID)
ON DELETE CASCADE
);
