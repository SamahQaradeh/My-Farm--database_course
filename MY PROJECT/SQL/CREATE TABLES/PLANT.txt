create table plant (
PLANT_ID NUMBER(20) NOT NULL,
PLANT_NAME VARCHAR2(30),
QUANTITY NUMBER(20),
SALARY NUMBER(20),
PLANT_DEGREE NUMBER(20),
TOXICITY VARCHAR2(30),
E_ID NUMBER (20),
F_ID NUMBER (20),
PRIMARY KEY(PLANT_ID),
FOREIGN KEY(E_ID) REFERENCES EMPLOYEE(EMP_ID)
              ON DELETE CASCADE,
FOREIGN KEY(F_ID) REFERENCES FERTILIZIR(FERT_ID)
             ON DELETE CASCADE
            
             
);
