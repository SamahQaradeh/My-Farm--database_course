CREATE TABLE ANIMALPRODUCT(
AP_ID NUMBER(20) NOT NULL,
AP_NAME VARCHAR2(30),
QUANTITY NUMBER(20),
SALARY NUMBER(20),
A_ID NUMBER(20),
E_ID NUMBER(20),
PRIMARY KEY(AP_ID),
FOREIGN KEY (A_ID) REFERENCES ANIMAL (ANIMAL_ID)
ON DELETE CASCADE,
FOREIGN KEY (E_ID) REFERENCES EMPLOYEE (EMP_ID)
ON DELETE CASCADE
);
