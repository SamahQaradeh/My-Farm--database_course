CREATE TABLE PLANTPRODUCT(
PP_ID NUMBER(20) NOT NULL,
PP_NAME VARCHAR2(30),
QUANTITY NUMBER(20),
SALARY NUMBER(20),
P_ID NUMBER(20),
E_ID NUMBER(20),
PRIMARY KEY(PP_ID),
FOREIGN KEY (P_ID) REFERENCES PLANT (PLANT_ID)
ON DELETE CASCADE,
FOREIGN KEY (E_ID) REFERENCES EMPLOYEE (EMP_ID)
ON DELETE CASCADE
);
